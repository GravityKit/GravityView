375

387

393  $filter['operator'] = empty( $filter['operator'] ) ? 'contains' : $filter['operator'];

396  * @filter `gravityview_search_operator` Modify the search operator for the field (contains, is, isnot, etc)

409 
410  if ( ! empty( $criteria['search_criteria'][ $key ] ) ) {
411 
412  // Use date_create instead of new DateTime so it returns false if invalid date format.
413  $date = date_create( $criteria['search_criteria'][ $key ] );
414 
415  if ( $date ) {
416  // Gravity Forms wants dates in the `Y-m-d H:i:s` format.
417  $criteria['search_criteria'][ $key ] = $date->format( 'Y-m-d H:i:s' );
418  } else {

420  unset( $criteria['search_criteria'][ $key ] );

581  /**
582  * Wrapper for the GFFormsModel::matches_operation() method that adds additional comparisons, including:
583  * 'equals', 'greater_than_or_is', 'greater_than_or_equals', 'less_than_or_is', 'less_than_or_equals',
584  * and 'not_contains'
585  *
586  * @since 1.13 You can define context, which displays/hides based on what's being displayed (single, multiple, edit)
587  *
588  * @link http://docs.gravityview.co/article/252-gvlogic-shortcode
589  * @uses GFFormsModel::matches_operation
590  * @since 1.7.5
591  *
592  * @param string $val1 Left side of comparison
593  * @param string $val2 Right side of comparison
594  * @param string $operation Type of comparison
595  *
596  * @return bool True: matches, false: not matches
597  */
598  public static function matches_operation( $val1, $val2, $operation ) {
599 
600  $value = false;
601 
602  if( 'context' === $val1 ) {
603 
604  $matching_contexts = array( $val2 );
605 
606  // We allow for non-standard contexts.
607  switch( $val2 ) {
608  // Check for either single or edit
609  case 'singular':
610  $matching_contexts = array( 'single', 'edit' );
611  break;
612  // Use multiple as alias for directory for consistency
613  case 'multiple':
614  $matching_contexts = array( 'directory' );
615  break;
616  }
617 
618  $val1 = in_array( gravityview_get_context(), $matching_contexts ) ? $val2 : false;
619  }
620 
621  switch ( $operation ) {
622  case 'equals':
623  $value = GFFormsModel::matches_operation( $val1, $val2, 'is' );
624  break;
625  case 'greater_than_or_is':
626  case 'greater_than_or_equals':
627  $is = GFFormsModel::matches_operation( $val1, $val2, 'is' );
628  $gt = GFFormsModel::matches_operation( $val1, $val2, 'greater_than' );
629  $value = ( $is || $gt );
630  break;
631  case 'less_than_or_is':
632  case 'less_than_or_equals':
633  $is = GFFormsModel::matches_operation( $val1, $val2, 'is' );
634  $gt = GFFormsModel::matches_operation( $val1, $val2, 'less_than' );
635  $value = ( $is || $gt );
636  break;
637  case 'not_contains':
638  $contains = GFFormsModel::matches_operation( $val1, $val2, 'contains' );
639  $value = ! $contains;
640  break;
641  default:
642  $value = GFFormsModel::matches_operation( $val1, $val2, $operation );
643  }
644 
645  return $value;
646  }
