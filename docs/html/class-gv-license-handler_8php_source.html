255  $is_ajax = ( defined('DOING_AJAX') && DOING_AJAX );
256  $data = empty( $array ) ? $_POST['data'] : $array;
257 
258  if ( $is_ajax && empty( $data['license'] ) ) {
259  die( - 1 );

261

263  $license_data = $this->_license_get_remote_response( $data, $license );
264 
265  // Empty is returned when there's an error.
266  if ( empty( $license_data ) ) {
267  if ( $is_ajax ) {
268  exit( json_encode( array() ) );
269  } else { // Non-ajax call
270  return json_encode( array() );
271  }
272  }
273 
274  $license_data->message = $this->get_license_message( $license_data );
275 
276  $json = json_encode( $license_data );
277 
278  $update_license = ( !isset( $data['update'] ) || !empty( $data['update'] ) );
279 
280  $is_check_action_button = ( 'check_license' === $data['edd_action'] && defined('DOING_AJAX') && DOING_AJAX );
281 
282  // Failed is the response from trying to de-activate a license and it didn't work.
283  // This likely happened because people entered in a different key and clicked "Deactivate",
284  // meaning to deactivate the original key. We don't want to save this response, since it is
285  // most likely a mistake.
286  if ( $license_data->license !== 'failed' && !$is_check_action_button && $update_license ) {
287 
288  if( !empty( $data['field_id'] ) ) {
289  set_transient( 'gravityview_' . esc_attr( $data['field_id'] ) . '_valid', $license_data, DAY_IN_SECONDS );
290  }
291 
292  $this->license_call_update_settings( $license_data, $data );
293 
294  }
295 
296  if ( $is_ajax ) {
297  exit( $json );
