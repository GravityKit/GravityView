700  /**
701  * Enable overrides of GravityView templates on a granular basis
702  *
703  * The loading order is:
704  *
705  * - view-[View ID]-table-footer.php
706  * - form-[Form ID]-table-footer.php
707  * - page-[ID of post or page where view is embedded]-table-footer.php
708  * - table-footer.php
709  *
710  * @see Gamajo_Template_Loader::get_template_file_names() Where the filter is
711  * @param array $templates Existing list of templates.
712  * @param [type] $slug [description]
713  * @param [type] $name [description]
714  */
715  function add_id_specific_templates( $templates, $slug, $name ) {
716 
717  $additional = array();
718 
719  // form-19-table-body.php
720  $additional[] = sprintf( 'form-%d-%s-%s.php', $this->getFormId(), $slug, $name );
721 
722  // view-3-table-body.php
723  $additional[] = sprintf( 'view-%d-%s-%s.php', $this->getViewId(), $slug, $name );
724 
725  if( $this->getPostId() ) {
726 
727  // page-19-table-body.php
728  $additional[] = sprintf( 'page-%d-%s-%s.php', $this->getPostId(), $slug, $name );
729  }
730 
731  // Combine with existing table-body.php and table.php
732  $templates = array_merge( $additional, $templates );
733 
734  do_action( 'gravityview_log_debug', '[add_id_specific_templates] List of Template Files', $templates );
735 
736  return $templates;
737  }
