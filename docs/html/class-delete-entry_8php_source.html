302  *
303  * 1. Check referrer validity
304  * 2. Make sure there's an entry with the slug of $_GET['entry_id']
305  * 3. If so, attempt to delete the entry. If not, set the error status
306  * 4. Remove `action=delete` from the URL
307  * 5. Redirect to the page using `wp_safe_redirect()`
308  *
309  * @since 1.5.1
310  * @uses GFAPI::delete_entry()
311  * @uses wp_safe_redirect()
312  * @return void
313  */
314  function process_delete() {
315 
316  // If the form is submitted
317  if( RGForms::get("action") === "delete") {
318 
319  $nonce_key = self::get_nonce_key( $_GET['entry_id'] );

323 
324  // Get the entry slug
325  $entry_slug = esc_attr( $_GET['entry_id'] );

327  // See if there's an entry there

329

334  if( is_wp_error( $has_permission ) ) {
335 
336  $messages = array(

338  'status' => 'error',
