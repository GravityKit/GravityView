139  * It also mimics the URL structure Gravity Forms expects to have so that

142  *

145  * @param $post_id int GravityView Post ID where View may be embedded {@since 1.9.2}
146  * @param string|array $field_values Parameters to pass in to the Edit Entry form to prefill data. Uses the same format as Gravity Forms "Allow field to be populated dynamically" {@since 1.9.2} {@see https://www.gravityhelp.com/documentation/article/allow-field-to-be-populated-dynamically/ }
147  * @return string
148  */
149  static function get_edit_link( $entry, $view_id, $post_id = null, $field_values = '' ) {
150 
151  $nonce_key = self::get_nonce_key( $view_id, $entry['form_id'], $entry['id'] );
152 
153  $base = gv_entry_link( $entry, $post_id );
154 
155  $url = add_query_arg( array(
156  'page' => 'gf_entries', // Needed for GFForms::get_page()
157  'view' => 'entry', // Needed for GFForms::get_page()
158  'edit' => wp_create_nonce( $nonce_key )
159  ), $base );
160 
161  /**
162  * Allow passing params to dynamically populate entry with values
163  * @since 1.9.2
164  */
165  if( !empty( $field_values ) ) {
166 
167  if( is_array( $field_values ) ) {
168  // If already an array, no parse_str() needed
169  $params = $field_values;
170  } else {
171  parse_str( $field_values, $params );
172  }
173 
174  $url = add_query_arg( $params, $url );
175  }
176 
177  return $url;

223  public static function check_user_cap_edit_entry( $entry, $view_id = 0 ) {
224 
225  // No permission by default
226  $user_can_edit = false;
227 
228  // Or if they can edit any entries (as defined in Gravity Forms), we're good.
229  if( GFCommon::current_user_can_any( 'gravityforms_edit_entries' ) ) {
230 
231  $user_can_edit = true;
232 
233  } else if( !isset( $entry['created_by'] ) ) {
234 
235  do_action('gravityview_log_error', 'GravityView_Edit_Entry[check_user_cap_edit_entry] Entry `created_by` doesn\'t exist.');
236 
237  $user_can_edit = false;
238 
239  } else {
240 
241  // get user_edit setting
242  if( empty( $view_id ) || $view_id == GravityView_View::getInstance()->getViewId() ) {
243  // if View ID not specified or is the current view
244  $user_edit = GravityView_View::getInstance()->getAtts('user_edit');
245  } else {
246  // in case is specified and not the current view
247  $user_edit = GVCommon::get_template_setting( $view_id, 'user_edit' );
248  }
249 
250  $current_user = wp_get_current_user();
251 
252  // User edit is disabled
253  if( empty( $user_edit ) ) {
254 
255  do_action('gravityview_log_debug', 'GravityView_Edit_Entry[check_user_cap_edit_entry] User Edit is disabled. Returning false.' );
256 
257  $user_can_edit = false;
258  }
259 
260  // User edit is enabled and the logged-in user is the same as the user who created the entry. We're good.
261  else if( is_user_logged_in() && intval( $current_user->ID ) === intval( $entry['created_by'] ) ) {
262 
263  do_action('gravityview_log_debug', sprintf( 'GravityView_Edit_Entry[check_user_cap_edit_entry] User %s created the entry.', $current_user->ID ) );
264 
265  $user_can_edit = true;
266  }
267 
268  }
269 
270  /**
271  * @param boolean $user_can_edit Can the current user edit the current entry? (Default: false)
272  */
273  $user_can_edit = apply_filters( 'gravityview/edit_entry/user_can_edit_entry', $user_can_edit );
274 
275  return (bool)$user_can_edit;
